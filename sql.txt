DROP DATABASE ME;
CREATE DATABASE ME;
USE ME;
CREATE TABLE USUARIOS(
	id MEDIUMINT AUTO_INCREMENT,
	NOMBRE CHAR(255),
	CORREO CHAR(255),
	CONTRASENA CHAR(255),
	PRIMARY KEY(id)
);
CREATE TABLE MENSAJE(
	id_mensaje MEDIUMINT AUTO_INCREMENT,
	id_receptor char(255),
	id_transmisor char(255),
	MENSAJE CHAR(255),
	PRIMARY KEY(id_mensaje)
);
